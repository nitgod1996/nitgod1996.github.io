/** * 先两两相加，再算2个数组可能的情况 * 时间复杂度O(N^2) * @param {number[]} nums1 * @param {number[]} nums2 * @param {number[]} nums3 * @param {number[]} nums4 * @return {number} */ var fourSumCount = function (nums1, nums2, nums3, nums4) { let map1 = new Map(), map2 = new Map(); // 因为所有数组都一样长，所以可以写在一个循环内 for (let i = 0; i < nums1.length; i++) { for (let j = 0; j < nums1.length; j++) { map1.set(nums1[i] + nums2[j], map1.get(nums1[i] + nums2[j]) + 1 || 1); map2.set(nums3[i] + nums4[j], map2.get(nums3[i] + nums4[j]) + 1 || 1); } } let sum = 0; for (let num of map1.keys()) { if (map2.has(-num)) { sum += map1.get(num) * map2.get(-num); } } return sum; }; console.log(fourSumCount([-1, -1], [-1, 1], [-1, 1], [1, -1]));
